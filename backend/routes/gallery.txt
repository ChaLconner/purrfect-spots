@router.get("/{photo_id}", response_model=CatLocation)
@limiter.limit("60/minute")
async def get_photo(
    request: Request,
    photo_id: str,
    gallery_service: GalleryService = Depends(get_gallery_service)
):
    """
    Get a specific photo by its ID.
    Useful for deep linking and sharing.
    """
    try:
        photo = gallery_service.get_photo_by_id(photo_id)
        if not photo:
            raise HTTPException(status_code=404, detail="Photo not found")
        
        return CatLocation(**photo)
    except HTTPException:
        raise
    except Exception as e:
        logger.error(f"Error fetching photo {photo_id}: {str(e)}", exc_info=True)
        raise HTTPException(
            status_code=500, detail="Failed to fetch photo"
        )
